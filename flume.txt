#exporting data to hdfs from flume(file name mysqld.log)
cd /var/log/
tail -f mysqld.log



cd /etc/flume-ng/conf
vi flume.conf


flume.conf contents
-------------
agent.channels=memory-channel
agent.sources=tail-source
agent.sinks=hdfs-sink
agent.sources.tail-source.type=exec
agent.sources.tail-source.command=tail -F /var/log/mysqld.log
agent.sources.tail-source.channels=memory-channel
agent.channels.memory-channel.type=memory
agent.sinks.hdfs-sink.channel=memory-channel
agent.sinks.hdfs-sink.type=hdfs
agent.sinks.hdfs-sink.hdfs.path=hdfs://localhost:8020/tmp/logs
agent.sinks.hdfs-sink.hdfs.fileType=DataStream


#delete if any logg file existitnf and create new
 hdfs dfs -mkdir /tmp/logs


 flume-ng agent --conf /etc/flume-ng/conf -f /etc/flume-ng/conf/flume.conf -Dflume.root.logger, console -n agent

service mysqld restart
hdfs dfs -ls /tmp/logs
